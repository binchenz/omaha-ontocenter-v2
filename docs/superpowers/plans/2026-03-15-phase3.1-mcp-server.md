# Phase 3.1 MCP Server Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Expose the Ontology layer as an MCP Server with 7 tools and API Key authentication, enabling external agents (Claude Desktop, Cursor, etc.) to query project data.

**Architecture:** A new `backend/app/mcp/` package implements the MCP Server using the Python MCP SDK. API Keys are stored hashed in a new `project_api_keys` table. The MCP tools delegate directly to the existing `omaha_service` — no business logic is duplicated.

**Tech Stack:** Python MCP SDK (`mcp`), FastAPI, SQLAlchemy, SQLite, `hashlib` (SHA256)

"""API Key model for MCP Server authentication."""
from sqlalchemy import Column, Integer, String, Boolean, DateTime, ForeignKey
from sqlalchemy.sql import func
from sqlalchemy.orm import relationship
from app.database import Base


class ProjectApiKey(Base):
    __tablename__ = "project_api_keys"

    id = Column(Integer, primary_key=True, index=True)
    project_id = Column(Integer, ForeignKey("projects.id"), nullable=False)
    name = Column(String, nullable=False)
    key_hash = Column(String, nullable=False, unique=True)   # SHA256(original_key)
    key_prefix = Column(String, nullable=False)              # random part first 8 chars
    is_active = Column(Boolean, default=True, nullable=False)
    expires_at = Column(DateTime(timezone=True), nullable=True)
    created_by = Column(Integer, ForeignKey("users.id"), nullable=False)
    created_at = Column(DateTime(timezone=True), server_default=func.now())

    project = relationship("Project", backref="api_keys")
    creator = relationship("User", backref="created_api_keys")

# backend/app/api/api_keys.py
"""API Key management endpoints."""
import hashlib
import secrets
from datetime import datetime
from typing import List, Optional
from fastapi import APIRouter, Depends, HTTPException, status
from pydantic import BaseModel
from sqlalchemy.orm import Session

from app.database import get_db
from app.models.user import User
from app.models.api_key import ProjectApiKey
from app.api.deps import get_current_user, get_project_for_owner

router = APIRouter()


class ApiKeyCreate(BaseModel):
    name: str
    expires_at: Optional[datetime] = None


class ApiKeyResponse(BaseModel):
    id: int
    name: str
    key_prefix: str
    is_active: bool
    expires_at: Optional[datetime]
    created_at: datetime

    model_config = {"from_attributes": True}


class ApiKeyCreated(ApiKeyResponse):
    key: str  # only returned once at creation


def _hash_key(key: str) -> str:
    return hashlib.sha256(key.encode()).hexdigest()


@router.post("/{project_id}/api-keys", response_model=ApiKeyCreated,
             status_code=status.HTTP_201_CREATED)
def create_api_key(
    project_id: int,
    body: ApiKeyCreate,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user),
):
    """Generate a new API key for a project."""
    get_project_for_owner(project_id, current_user, db)
    random_part = secrets.token_hex(16)  # 32 chars
    key = f"omaha_{project_id}_{random_part}"
    api_key = ProjectApiKey(
        project_id=project_id,
        name=body.name,
        key_hash=_hash_key(key),
        key_prefix=random_part[:8],
        expires_at=body.expires_at,
        created_by=current_user.id,
    )
    db.add(api_key)
    db.commit()
    db.refresh(api_key)
    return {**ApiKeyResponse.model_validate(api_key).model_dump(), "key": key}


@router.get("/{project_id}/api-keys", response_model=List[ApiKeyResponse])
def list_api_keys(
    project_id: int,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user),
):
    """List all API keys for a project."""
    get_project_for_owner(project_id, current_user, db)
    return db.query(ProjectApiKey).filter(
        ProjectApiKey.project_id == project_id
    ).all()


@router.delete("/{project_id}/api-keys/{key_id}", status_code=status.HTTP_204_NO_CONTENT)
def revoke_api_key(
    project_id: int,
    key_id: int,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user),
):
    """Revoke (deactivate) an API key."""
    get_project_for_owner(project_id, current_user, db)
    api_key = db.query(ProjectApiKey).filter(
        ProjectApiKey.id == key_id,
        ProjectApiKey.project_id == project_id,
    ).first()
    if not api_key:
        raise HTTPException(status_code=404, detail="API key not found")
    api_key.is_active = False
    db.commit()
```

# backend/app/mcp/auth.py
"""API Key authentication for MCP Server."""
import hashlib
import os
from datetime import datetime, timezone
from typing import Optional, Tuple
from sqlalchemy.orm import Session
from app.database import SessionLocal
from app.models.api_key import ProjectApiKey
from app.models.project import Project


def _hash_key(key: str) -> str:
    return hashlib.sha256(key.encode()).hexdigest()


def resolve_api_key(key: str) -> Optional[Tuple[int, str]]:
    """
    Validate API key and return (project_id, omaha_config) or None.
    Checks: exists, is_active, not expired.
    """
    db: Session = SessionLocal()
    try:
        api_key = db.query(ProjectApiKey).filter(
            ProjectApiKey.key_hash == _hash_key(key),
            ProjectApiKey.is_active == True,
        ).first()

        if not api_key:
            return None

        if api_key.expires_at and api_key.expires_at < datetime.now(timezone.utc):
            return None

        project = db.query(Project).filter(Project.id == api_key.project_id).first()
        if not project or not project.omaha_config:
            return None

        return (project.id, project.omaha_config)
    finally:
        db.close()


def get_api_key_from_env() -> str:
    """Get API key from environment variable (stdio mode)."""
    key = os.environ.get("OMAHA_API_KEY", "")
    if not key:
        raise ValueError("OMAHA_API_KEY environment variable not set")
    return key
```

# backend/app/mcp/tools.py
"""MCP tool implementations — delegates to omaha_service."""
from typing import Any, Dict, List, Optional
from app.services.omaha import omaha_service
from app.database import SessionLocal
from app.models.asset import DatasetAsset, DataLineage


def list_objects(config_yaml: str) -> Dict[str, Any]:
    """List all Ontology object types in the project."""
    result = omaha_service.build_ontology(config_yaml)
    if not result.get("valid"):
        return {"error": "Invalid configuration"}
    objects = result.get("ontology", {}).get("objects", [])
    return {"objects": [obj.get("name") for obj in objects if obj.get("name")]}


def get_schema(config_yaml: str, object_type: str) -> Dict[str, Any]:
    """Get field definitions for an object type."""
    return omaha_service.get_object_schema(config_yaml, object_type)


def get_relationships(config_yaml: str, object_type: str) -> Dict[str, Any]:
    """Get relationships for an object type."""
    rels = omaha_service.get_relationships(config_yaml, object_type)
    return {"relationships": rels}


def query_data(
    config_yaml: str,
    object_type: str,
    selected_columns: Optional[List[str]] = None,
    filters: Optional[List[Dict[str, Any]]] = None,
    joins: Optional[List[Dict[str, Any]]] = None,
    limit: int = 100,
) -> Dict[str, Any]:
    """Execute a query against the ontology."""
    return omaha_service.query_objects(
        config_yaml, object_type, selected_columns, filters, joins, limit
    )


def save_asset(
    project_id: int,
    name: str,
    base_object: str,
    selected_columns: Optional[List[str]] = None,
    filters: Optional[List[Dict[str, Any]]] = None,
    joins: Optional[List[Dict[str, Any]]] = None,
    description: Optional[str] = None,
    row_count: Optional[int] = None,
) -> Dict[str, Any]:
    """Save a query configuration as a reusable asset."""
    import json
    db = SessionLocal()
    try:
        asset = DatasetAsset(
            project_id=project_id,
            name=name,
            description=description,
            base_object=base_object,
            selected_columns=json.dumps(selected_columns or []),
            filters=json.dumps(filters or []),
            joins=json.dumps(joins or []),
            row_count=row_count,
            created_by=1,  # MCP context: use project owner
        )
        db.add(asset)
        db.commit()
        db.refresh(asset)
        return {"id": asset.id, "name": asset.name, "message": "Asset saved"}
    finally:
        db.close()


def list_assets(project_id: int) -> Dict[str, Any]:
    """List saved assets for the project."""
    db = SessionLocal()
    try:
        assets = db.query(DatasetAsset).filter(
            DatasetAsset.project_id == project_id
        ).all()
        return {"assets": [
            {"id": a.id, "name": a.name, "base_object": a.base_object,
             "row_count": a.row_count, "created_at": str(a.created_at)}
            for a in assets
        ]}
    finally:
        db.close()


def get_lineage(asset_id: int) -> Dict[str, Any]:
    """Get data lineage for an asset."""
    db = SessionLocal()
    try:
        records = db.query(DataLineage).filter(
            DataLineage.asset_id == asset_id
        ).all()
        return {"lineage": [
            {"type": r.lineage_type, "source": r.source_name,
             "source_type": r.source_type}
            for r in records
        ]}
    finally:
        db.close()
```

# backend/app/mcp/server.py
"""MCP Server entry point — stdio mode."""
import asyncio
import json
from mcp.server import Server
from mcp.server.stdio import stdio_server
from mcp import types

from app.mcp.auth import get_api_key_from_env, resolve_api_key
from app.mcp import tools as t

app = Server("omaha-ontocenter")


def _get_context():
    """Resolve API key and return (project_id, config_yaml)."""
    key = get_api_key_from_env()
    result = resolve_api_key(key)
    if not result:
        raise ValueError("Invalid or expired API key")
    return result


@app.list_tools()
async def list_tools() -> list[types.Tool]:
    return [
        types.Tool(name="list_objects", description="List all Ontology object types",
                   inputSchema={"type": "object", "properties": {}}),
        types.Tool(name="get_schema", description="Get field definitions for an object type",
                   inputSchema={"type": "object", "properties": {
                       "object_type": {"type": "string"}}, "required": ["object_type"]}),
        types.Tool(name="get_relationships", description="Get relationships for an object type",
                   inputSchema={"type": "object", "properties": {
                       "object_type": {"type": "string"}}, "required": ["object_type"]}),
        types.Tool(name="query_data", description="Execute a query against the ontology",
                   inputSchema={"type": "object", "properties": {
                       "object_type": {"type": "string"},
                       "selected_columns": {"type": "array", "items": {"type": "string"}},
                       "filters": {"type": "array"},
                       "joins": {"type": "array"},
                       "limit": {"type": "integer", "default": 100}},
                   "required": ["object_type"]}),
        types.Tool(name="save_asset", description="Save a query as a reusable asset",
                   inputSchema={"type": "object", "properties": {
                       "name": {"type": "string"},
                       "base_object": {"type": "string"},
                       "description": {"type": "string"},
                       "selected_columns": {"type": "array"},
                       "filters": {"type": "array"},
                       "joins": {"type": "array"},
                       "row_count": {"type": "integer"}},
                   "required": ["name", "base_object"]}),
        types.Tool(name="list_assets", description="List saved assets",
                   inputSchema={"type": "object", "properties": {}}),
        types.Tool(name="get_lineage", description="Get data lineage for an asset",
                   inputSchema={"type": "object", "properties": {
                       "asset_id": {"type": "integer"}}, "required": ["asset_id"]}),
    ]


@app.call_tool()
async def call_tool(name: str, arguments: dict) -> list[types.TextContent]:
    project_id, config_yaml = _get_context()

    if name == "list_objects":
        result = t.list_objects(config_yaml)
    elif name == "get_schema":
        result = t.get_schema(config_yaml, arguments["object_type"])
    elif name == "get_relationships":
        result = t.get_relationships(config_yaml, arguments["object_type"])
    elif name == "query_data":
        result = t.query_data(config_yaml, **arguments)
    elif name == "save_asset":
        result = t.save_asset(project_id, **arguments)
    elif name == "list_assets":
        result = t.list_assets(project_id)
    elif name == "get_lineage":
        result = t.get_lineage(arguments["asset_id"])
    else:
        result = {"error": f"Unknown tool: {name}"}

    return [types.TextContent(type="text", text=json.dumps(result, ensure_ascii=False))]


async def main():
    async with stdio_server() as (read_stream, write_stream):
        await app.run(read_stream, write_stream, app.create_initialization_options())


if __name__ == "__main__":
    asyncio.run(main())
```
